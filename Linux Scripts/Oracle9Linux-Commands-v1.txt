# List all active services on Oracle Linux 9 (systemd-based):
systemctl list-units --type=service --state=running

# Specifically check service SSHD (SSH/SFTP):
systemctl status sshd

# Check CPU & Memory Load:
Top

# Check Uptime
Uptime

# Check sshd service status
systemctl status sshd

# Show detailed SSH proccesses 
ps -ef | grep sshd

# Show all SSH sessions (including SFTP)
ss -tnp | grep sshd

# Check active SFTP sessions:
ps -ef | grep internal-sftp

# Check for zombie processes and their parent
ps -eo pid,ppid,state,cmd | grep 'Z'

# Check disk usage with:
df -h

# Show logged-in users:
who
Or
w

# Reload SSHD Service. Reloads the configuration without killing active sessions. However, if the service is stuck, reload may not fix it.
sudo systemctl reload sshd

# Restart SSHD Service.
sudo systemctl restart sshd
